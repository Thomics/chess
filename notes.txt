function createBoard() {
  var chessSquare;
  var chessRow = [];
  var squareNum = 0;
  var text = '';

  for ( var i = 0; i <= 7; i++ ) {
    for ( var j = 0; j <= 7; j++ ) {
      if ( i <= 1 || i >= 6 ) {
        chessSquare = setPieces(i, j, squareNum);
      } else {
        chessSquare = new vm.Square(i, j, squareNum, false, {row: i, column: i, pieceLetter: ''});
      }
      chessRow.push(chessSquare);
      text += vm.generateHtml(chessSquare);
      squareNum++;
    }
    vm.chessboard.push(chessRow);
    chessRow = [];
  }
  document.getElementById('chessboard-container').innerHTML = text;
}
